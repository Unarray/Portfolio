Utilitaire logger

  • Permet une meilleure cohérence dans la codebase

Exemples/demos

Logger global pour les infos, succès et erreurs

var logger = log.New(os.Stdout, "", 0)

func Info(message string) {
	logger.Println(formatLog(Yellow, "info", message))
}

func Success(message string) {
	logger.Println(formatLog(Green, "succes", message))
}

func Error(message string, execStack bool) {
	logger.Println(formatLog(Red, "error", message))
	if execStack {
		debug.PrintStack()
	}
}

func Fatal(message string) {
	Error(message, true)
	os.Exit(1)
}

func formatLog(color, logType, message string) string {
	maxSpace := 8
	spaceSize := maxSpace - len(logType)
	logType = strings.ToUpper(logType)

	typePrefix := fmt.Sprintf("%v%v%v", Bold, color, logType)
	separator := fmt.Sprintf("%v %v%v» ", Reset, Gray, strings.Repeat("-", spaceSize))
	content := fmt.Sprintf("%v%v", Reset, message)

	return BaseLogFormat(typePrefix + separator + content)
}

func BaseLogFormat(message string) string {
	datetime := time.Now().Format("02/01/2006 15:04:05.000")

	return fmt.Sprintf("%v[%v] %v", White, datetime, message)
}

Un logger plus spécifique pour log les requêtes entrantes/sortantes sur notre API

func InRequest(requestId, method, route string) {
	logger.Println(formatRequestLog(requestId, method, route, true))
}

func OutRequest(requestId, method, route string, status int, elapsedTime time.Duration) {
	logger.Printf(
		"%v - %v%v%v (%v)",
		formatRequestLog(requestId, method, route, false),
		statusColor(status), status, Default,
		elapsedTime.String(),
	)
}

func formatRequestLog(requestId, method, route string, isRequestIn bool) string {
	var requestTarget string

	if isRequestIn {
		requestTarget = "=>"
	} else {
		requestTarget = "<="
	}

	return BaseLogFormat(fmt.Sprintf(
		"%v%v%v %v %v%v %v%v",
		Underscore, requestId, Reset,
		requestTarget,
		methodColor(method), method,
		Default, route,
	))
}

func methodColor(method string) string {
	switch method {
	case fiber.MethodGet:
		return Cyan
	case fiber.MethodPost:
		return Green
	case fiber.MethodPut:
		return Yellow
	case fiber.MethodDelete:
		return Red
	default:
		return Reset
	}
}

func statusColor(code int) string {
	switch {
	case code >= fiber.StatusOK && code < fiber.StatusMultipleChoices:
		return Green
	case code >= fiber.StatusMultipleChoices && code < fiber.StatusBadRequest:
		return Blue
	case code >= fiber.StatusBadRequest && code < fiber.StatusInternalServerError:
		return Yellow
	default:
		return Red
	}
}

func LoadVariables() {
	err := godotenv.Load()

	if err != nil {
		logger.Fatal("Error loading .env file")
	}

	values := reflect.ValueOf(&Variables).Elem()
	types := values.Type()

	for i := 0; i < types.NumField(); i++ {
		field := types.Field(i)
		value := os.Getenv(field.Name)

		values.Field(i).Set(reflect.ValueOf(value))
	}

	if errors := common.ValidateStruct(&Variables); errors != nil {
		logger.Fatal(fmt.Sprintf("%v", errors))
	}

	logger.Success("environement variables loaded")
}

func CreateAccount(email, name, last_name, password string) (*models.Account, bool) {
	account := models.Account{
		Email:    email,
		Password: password,
		LastName: last_name,
		Name:     name,
	}

	if err := database.Instance.Create(&account).Error; err != nil {
		if err == gorm.ErrDuplicatedKey {
			return nil, true
		}

		logger.Error("Database `CreateAccount`: "+err.Error(), false)
		return nil, false
	}

	logger.Info("Database `CreateAccount`: " + account.Email + " created")
	return &account, true
}
Le password est hashé avant que cette fonction soit appelé
L'algorithme de hashage utilisé est Argon2

Lors de la création d'un compte, l'utilisateur passe par plusieurs étapes

  1. Création du compte
    L'Email, mots de passe, nom, prénom est envoyé au backend

    Le backend va créer inséré sur REDIS un token associé aux informations du compte a créer.
    A cette endroit, le mots de passe est hashé pour qu'il ne soit pas stocké en claire sur REDIS

  2. Validation par Email
    L'utilisateur reçoit par E-Mail un lien "magique" lui permettant de valider la création de son compte

    Le backend va vérifier si le token est correct, si c'est le cas, il va créer un nouvelle utilisateur sur la base de données avec les informations stocké sur REDIS, puis supprimer les informations de REDIS

func GetAccount(accountID string, preloadFields []string) (*models.Account, bool) {
	account := models.Account{
		Email: accountID,
	}

	if err := database.PreloadFields(database.Instance, preloadFields).First(&account).Error; err != nil {
		if err == gorm.ErrRecordNotFound {
			return nil, true
		}

		logger.Error("Database `GetAccount`: "+err.Error(), false)
		return nil, false
	}

	return &account, true
}

Anciennes fonctions SSO
package cache

import (
	"Seedext/src/database"
	"context"
	"math/rand"
	"time"
)

const ssoPrefix = "sso-"

func GenerateSingleSignOnCode(accountId string) int64 {
	genDigit := func() int64 { return int64(rand.Intn(8) + 1) }
	code := genDigit() + 10*genDigit() + 100*genDigit() + 1000*genDigit()

	database.CacheInstance.Set(context.Background(), ssoPrefix+accountId, code, time.Minute*5)
	return code
}

func TryGetSingleSignOnCode(accountId string) int64 {
	code, e := database.CacheInstance.Get(context.Background(), ssoPrefix+accountId).Int64()
	if e != nil {
		return -1
	}

	return code
}

La gestion des erreurs n'était pas faite, des logique manquaient (par exemple si un utilisateur demande plusieurs fois un code)

Sur ce refactor, les fonctions gèrent les erreurs, et prennent en compte cette logique (tout en utilisant le logger pour pouvoir track l'origine d'une errerur)

Nouvelles fonctions SSO
package methods

import (
	"Seedext/src/cache"
	"Seedext/src/utils/logger"
	"Seedext/src/utils/random"
	"context"
	"time"

	"github.com/redis/go-redis/v9"
)

const ssoCodePrefix = "sso_code-"
const ssoAccountPrefix = "sso_account-"

func GenerateSingleSignOnCode(accountID string) (string, bool) {
	code, success := GetSingleSignOnCode(accountID)

	if !success {
		return "", false
	}

	const expirationTime = time.Minute * 5

	if code != "" {
		pipe := cache.Instance.Pipeline()

		pipe.Expire(context.Background(), ssoCodePrefix+code, expirationTime)
		pipe.Expire(context.Background(), ssoAccountPrefix+accountID, expirationTime)

		_, err := pipe.Exec(context.Background())

		if err != nil {
			logger.Error("Cache `GenerateSingleSignOnCode`: "+err.Error(), false)
			return "", false
		}

		return code, true
	}

	ssoCode := random.RandomNumber(6)
	pipe := cache.Instance.Pipeline()

	pipe.Set(
		context.Background(),
		ssoCodePrefix+ssoCode,
		accountID,
		expirationTime,
	)
	pipe.Set(
		context.Background(),
		ssoAccountPrefix+accountID,
		ssoCode,
		expirationTime,
	)

	_, err := pipe.Exec(context.Background())

	if err != nil {
		logger.Error("Cache `GenerateSingleSignOnCode`: "+err.Error(), false)
		return "", false
	}

	return ssoCode, true
}

func getSingleSignOnAccount(code string) (string, bool) {
	accountID, err := cache.Instance.Get(context.Background(), ssoCodePrefix+code).Result()

	if err != nil {
		if err == redis.Nil {
			return "", true
		}

		logger.Error("Cache `getSingleSignOnAccount`: "+err.Error(), false)
		return "", false
	}

	return accountID, true
}

func GetSingleSignOnCode(accountID string) (string, bool) {
	code, err := cache.Instance.Get(context.Background(), ssoAccountPrefix+accountID).Result()

	if err != nil {
		if err == redis.Nil {
			return "", true
		}

		logger.Error("Cache `GetSingleSignOnCode`: "+err.Error(), false)
		return "", false
	}

	return code, true
}

func RemoveSingleSignOnCode(code string) bool {
	accountID, success := getSingleSignOnAccount(code)

	if !success {
		return false
	}

	err := cache.Instance.Del(context.Background(), ssoCodePrefix+code, ssoAccountPrefix+accountID).Err()

	if err != nil {
		logger.Error("Cache `RemoveSingleSignOnCode`: "+err.Error(), false)
		return false
	}

	return true
}

Refactor du package routes

Ce package a quelques petites règles uniques. Premièrement le package a été renommé en api.
In contient le fichier router.go qui enregistre toute les routes de notre API.

Nous avons ensuite un package api/routes qui va contenir les routes de notre API.
Ce package a des règles uniques:

  • Chaque package dans routes représente une route
  • La méthode utiliser pour la route est représenté par un fichier représentant son nom (par exemple pour la route POST /account/settings, le fichier sera: /src/api/routes/account/settings/post.go avec une fonction func Post() a l'intérieur)
    Pour respecter les conventions RESTful, les uniques méthodes HTTP que nous allons utiliser sont:
    • POST → Créer une donnée (CRUD)
    • GET → Récupérer une donnée (CRUD)
    • PUT → Mettre a jour une donnée (CRUD)
    • DELETE → Supprimer une donnée (CRUD)
  • Les paramètre des routes dynamiques son représenté par un des underscore _ entourant le nom du dossier (par exemple pour la route DELETE /meetings/:id le chemin sera /src/api/routes/meetings/_id_/delete.go)
    Ainsi la structure de nos fichiers reflète nos routes API, ce qui rend plus clair la localisation des fichiers. Si nous voulons modifier l'endpoint GET /foo/bar/:titi/toto nous devons modifier la fonction Get localisée dans /src/api/routes/foo/bar/_titi_/toto/get.go.
    Cette structure a été inspiré du routing que propose le framework JavaScript Next.JS

Avec ces règles, j'ai donc refait les routes GET /ping, GET /account, POST /auth/sign/up, POST /auth/sign/up/confirm, POST /auth/sing/in et POST /auth/password/forgotten

Le fichier /src/api/routes/auth/sign/up/post.go ressemble donc a ceci:

Route POST /auth/sign/up
// Post godoc
//
//	@Summary			Register an account
//	@Description	    Start an account creation process
//	@Tags				auth
//	@Param				"Request params" body PostRequest true "Parameters to init an account creation"
//	@Accept				json
//	@Produce			json
//	@Success			202
//	@Failure			400,409,500,503	{object}	errors.ErrorResponse
//	@Router				/auth/sign/up [post]
func Post(c *fiber.Ctx) error {
	if !mail.Enabled {
		return c.Status(fiber.StatusServiceUnavailable).JSON(errors.GetErrorReponse(errors.GenericMailDisabled))
	}

	body := PostRequest{}

	if success, errorResponse := common.ValidateRequest(c, &body); !success {
		return c.Status(fiber.StatusBadRequest).JSON(errorResponse)
	}

	if !validator.VerifySeedextURL(body.FallbackURL) {
		return c.Status(fiber.StatusBadRequest).JSON(errors.GetErrorReponse(errors.GenericInvalidURL))
	}

	// Get account
	account, success := dbMethods.GetAccount(body.Email, []string{})

	if !success {
		return c.Status(fiber.StatusInternalServerError).JSON(errors.GetErrorReponse(errors.GenericInternalServerError))
	}

	if account != nil {
		return c.Status(fiber.StatusConflict).JSON(errors.GetErrorReponse(errors.AccountAlreadyExist))
	}

	// Hash the received password
	passwordHash, err := hash.GenerateHashFromPassword(body.Password)

	if err != nil {
		return c.Status(fiber.StatusInternalServerError).JSON(errors.GetErrorReponse(errors.GenericInternalServerError))
	}

	// Generate a TOKEN to confirm account creation
	signOnToken, success := cacheMethods.GenerateSignUpToken(body.Email, body.LastName, body.Name, passwordHash)

	if !success {
		return c.Status(fiber.StatusInternalServerError).JSON(errors.GetErrorReponse(errors.GenericInternalServerError))
	}

	// Get the EMAIL to send
	mailContent, success := mail.ParseMail(
		"./mails/out/button-message.html",
		map[mail.Tag]interface{}{
			mail.MailDescription: "Cliquez sur le bouton si dessous pour confirmer votre compte",
			mail.MailButtonText:  "Confirmer",
			mail.MailButtonLink:  utils.AddParamToURL(body.FallbackURL, "confirm_token", signOnToken),
		},
	)

	if !success {
		return c.Status(fiber.StatusServiceUnavailable).JSON(errors.GetErrorReponse(errors.GenericInternalServerError))
	}

	// Send the EMAIL
	success = mail.SendMail(body.Email, "Confirmez votre compte Seedext", mailContent)

	if !success {
		return c.Status(fiber.StatusServiceUnavailable).JSON(errors.GetErrorReponse(errors.GenericInternalServerError))
	}

	return c.SendStatus(fiber.StatusAccepted)
}

Pour celui-là c'est principalement des prompt a un LLM qui nous permet de traduire les titres de nos synthèses. Nos synthèses sont rédigées dans la même langue que notre transcription. Mais les titres ce sont nous qui les définitions. Pour qu'ils soit dans la même langue que la transcription. Nous faisons ce prompt:

You're a program for finding the language of a text. Your sole purpose is to give the user ONLY the ISO 639-1 code of the text without making sentence.

Avec ce prompt, ca nous permet de récupérer la langue du texte donné sous le format ISO 639-1. Le format ISO 639-1 sont des codes a deux lettres représentant le nom des langues.

Cette solution est temporaire. Plus tard nous utiliserons des librairies spécialisées pour cette tache (comme par exemple LinguaGO)

Ensuite nous avons juste a traduire nos titres avec le code ISO retourné:

func GetNChar(text string, n int) string {

	return text

Support d'upload volumineux

Toute les applications sont basés sur la Google Cloud Platform (GCP). Il faut savoir que pour les requetes HTTP/1 ont une limite de 32 Mio. Google recommande donc pour l'upload de fichiers volumineux d'utiliser les "SignedURL" pour upload un fichier sur le bucket sans passer par notre API.

Le problème est que nous envoyons ce fichiers vers notre service de transcription pour ensuite pouvoir faire les synthèses, etc...
Si notre fichier est upload directement sur le bucket Google Cloud Storage, comment lier notre API ?

C'est la qu'interviennent les PUB/SUB GCP.

Google Cloud Pub/Sub est un service de messagerie asynchrone évolutif offert par Google Cloud Platform. Il permet la communication entre les services de manière déconnectée, avec une latence d'environ 100 millisecondes. Ce service est conçu pour dissocier les producteurs de messages des consommateurs, ce qui favorise la scalabilité et la flexibilité des systèmes.

Plus d'informations ici

Un PUB/SUB est composé de "Topic" composé de "Subscriptions"
Une entité externe envoie un message a un Topic. Ce Topic va le transmettre a ses subscriptions ce message. Nous avons ensuite des entités abonnées qui vont récupérer les messages retenus dans les subscription


Une fois qu'un message a été traité avec succès par une entité, l'entité envoi un "Ack" (Acknowledgment ) et va retirer le message de la queue. (si le message n'est pas ack, il reste donc dans la queue)

Maintenant que nous connaissons le principe des PUB/SUB GCP, il faut envoyer des messages sur un Topic quand un fichier est téléversé sur notre bucket. Pour cella, Google Cloud Storage nous permet d'envoyer des messages a certains évènements (doc GCS PUB/SUB)

Type d'évènement Description
OBJECT_FINALIZE Envoyé lorsqu'un objet (ou une nouvelle génération d'un objet existant) est bien créé dans le bucket. Cela inclut la copie ou la réécriture d'un objet existant. Les échecs d'importation ne génèrent pas cet événement.
OBJECT_METADATA_UPDATE Envoyé lorsque les métadonnées d'un objet existant sont modifiées.
OBJECT_DELETE Envoyé lorsqu'un objet a été définitivement supprimé. Cela inclut les objets remplacés ou supprimés dans le cadre de la configuration du cycle de vie du bucket. Cela n'inclut pas les objets qui sont obsolètes (voir OBJECT_ARCHIVE) ni les importations en plusieurs parties annulées.
OBJECT_ARCHIVE Envoyé uniquement lorsque la gestion des versions d'objets est activée sur un bucket. Cet événement indique que la version en ligne d'un objet est devenue obsolète (soit parce qu'elle a été explicitement rendue obsolète, soit parce qu'elle a été remplacée lors de l'importation d'un objet portant le même nom).

Pour créer ces notifications, nous devons éxecuter cette commande:

gcloud storage buckets notifications create gs://BUCKET_NAME --topic=TOPIC_NAME

On peut choisir les évènements en utilisant le flag --event-types=[NOTIFICATION_EVENT_TYPE,…] (ou --e=[NOTIFICATION_EVENT_TYPE,…])

Ainsi notre commande ressemble a:

gcloud storage buckets notifications create gs://seedext-backend-dev --topic=bucket -e="OBJECT_FINALIZE"

Avec cette configuration, notre bucket enverra une message sur le Topic "bucket" lorsqu'un nouvelle objet sera créé

Nous devons maintenant configurer un abonnement.
Nous avons le choix entre deux type de notification.

  • Pull - C'est le client qui demande le message d'un abonnement. Une fois traité il doit envoyer une ack
  • Push - C'est un webhook. L'abonnement s'occupe d'envoyer sur l'endpoint definie une requete post. Pour répondre un ack, nous devons envoyer un status code 2xx (tout autre status code ne validera pas le message)
    Dans notre cas, le type push est le plus adapté puisque nous ne voulons exécuter une action quand un fichier est upload sur notre bucket (un évènement imprévisible)

// GetUploadURL godoc
// @Summary Route to get an upload URL
// @Description Route to get an upload URL
// @Param extension query string true "File extension (ie: ".mp3")"
// @Tags meetings
// @Accept application/json
// @Produce json
// @Success 200 {object} GetUploadLinkResponse
// @Failure 400 {object} common.ErrorResponse
// @Router /meetings/upload_url [get]
func GetUploadURL(c *fiber.Ctx) error {
	extension := c.Query("extension")

	if extension == "" {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node:             "meetings.missing-extension",
			ValidationErrors: []common.ValidationError{},
		})
	}

	if !slices.Contains(valid_extention, strings.ToLower(extension)) {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node:             "meetings.unsupported-file-extension",
			ValidationErrors: []common.ValidationError{},
		})
	}

	fileID := utils.GenerateRandomUuid() + extension
	bucket := storage.ApiClient.Bucket(string(storage.MeetingsBucket))
	attr, err := bucket.Object(fileID).Attrs(context.Background())

	if attr != nil {
		return c.SendStatus(fiber.StatusInternalServerError)
	}

	if err != nil && err != googleStorage.ErrObjectNotExist {
		return c.SendStatus(fiber.StatusServiceUnavailable)
	}

	expiration := time.Now().Add(time.Minute)
	signedURL, err := bucket.GenerateSignedPostPolicyV4(fileID, &googleStorage.PostPolicyV4Options{
		Expires: expiration,
	})

	if err != nil {
		return c.SendStatus(fiber.StatusInternalServerError)
	}

	accountId := c.Locals("account_id").(string)

	success := cache.SetMeetingUpload(string(storage.MeetingsBucket)+"/"+fileID, accountId)

	if !success {
		return c.SendStatus(fiber.StatusInternalServerError)
	}

	return c.Status(fiber.StatusOK).JSON(GetUploadLinkResponse{
		signedURL.URL,
		signedURL.Fields,
	})
}

// GCP webhook.
func UploadedMeeting(c *fiber.Ctx) error {
	if !deepgram.Active {
		logger.Error("Tried to upload but module is disabled.", false)
		return c.SendStatus(fiber.StatusServiceUnavailable)
	}

	// Validate JSON Web Token
	authHeader := c.Get("Authorization")
	if authHeader == "" || len(strings.Split(authHeader, " ")) != 2 {
		return c.SendStatus(fiber.StatusUnauthorized)
	}

	token := strings.Split(authHeader, " ")[1]

	payload, err := idtoken.Validate(context.Background(), token, utils.GetLocalUrl()+"/api/meetings/uploaded_meeting")
	if err != nil {
		return c.SendStatus(fiber.StatusBadRequest)
	}

	if payload.Claims["email"] != storage.BucketEmail || payload.Claims["email_verified"] != true {
		return c.SendStatus(fiber.StatusBadRequest)
	}

	// Start parsing
	body := UploadedMeetingRequest{}

	errors := common.ValidateRequest(c, &body)
	if errors != nil {
		return errors
	}

	eventType, exist := body.Message.Attributes["eventType"]

	if !exist {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.event",
		})
	}

	if eventType != googleStorage.ObjectFinalizeEvent {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.wrong_event",
		})
	}

	bucketID, exist := body.Message.Attributes["bucketId"]

	if !exist {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.bucket",
		})
	}

	if bucketID != string(storage.MeetingsBucket) {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.wrong_bucket",
		})
	}

	fileID, exist := body.Message.Attributes["objectId"]

	if !exist {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.file",
		})
	}

	// IF FILE HAS ALREADY BEEN UPLOADED (recall webhook)
	exists, _ := files.FindFile(fileID)

	if exists {
		return c.Status(fiber.StatusOK).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.already_compute",
		})
	}

	exist, accountID := cache.GetMeetingUpload(bucketID + "/" + fileID)

	if !exist {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.cache_account",
		})
	}

	exist, _ = account.FindAccount(accountID)

	if !exist {
		return c.Status(fiber.StatusBadRequest).JSON(common.ErrorResponse{
			Node: "meetings.uploaded.account",
		})
	}

	signedURL, err := storage.ApiClient.Bucket(string(storage.MeetingsBucket)).SignedURL(
		fileID,
		&googleStorage.SignedURLOptions{
			Method:  "GET",
			Expires: time.Now().Add(time.Minute * 5),
		},
	)
	if err != nil {
		return c.SendStatus(fiber.StatusInternalServerError)
	}

	cache.DeleteMeetingUpload(bucketID + "/" + fileID)

	meetingID := uploadMeeting(fileID, signedURL, accountID)

	account.AddActivity(accountID, meetingID)
	account.AddMeeting(accountID, meetingID)

	return c.SendStatus(fiber.StatusOK)
}

Faire en sorte que les bots rejoignent automatiquement lorsque sont calendrier Google ou Microsoft est connecté a son compte

Pour cela, la version de l'API que nous utilisons pour les calendriers n'est pas très adaptée pour notre usage (elle sera modifier dans le refactor)
Pour cela, j'ai créé un CRON-JOB qui récupère les prochaines réunions de l'utilisateur.

La task
package calendar

import (
	"Seedext/src/database"
	"Seedext/src/database/cache"
	"Seedext/src/database/collections/account"
	"Seedext/src/services/recall"
	"Seedext/src/utils/logger"
	"context"
	"time"
)

func TrackUsersMeetingsTask() {
	if !recall.Active {
		logger.Info("TrackUsersMeetingsTask Task skipped (recall disabled)")
		return
	}

	accounts, success := account.FindAllAccountID()

	if !success {
		logger.Error("`TrackUsersMeetingsTask`: Failed to get users id", false)
		return
	}

	for _, account := range *accounts {

		exist, token := cache.GetCalendarAuthToken(account.AccountId)

		if !exist {
			response := recall.ApiClient.GetCalendarAuthToken(account.AccountId)
			token = response.Token

			cache.SetCalendarAuthToken(account.AccountId, token)
		}

		meetings := recall.ApiClient.ListCalendarMeetings(token)

		for _, meeting := range meetings {
			if meeting.OverrideShouldRecord || meeting.WillRecord {
				continue
			}

			fetchedMeeting := recall.ApiClient.UpdateCalendarMeeting(token, meeting.Id, true)
			bot := recall.ApiClient.RetrieveMeetingBot(fetchedMeeting.BotId)
			link := getMeetingLink(bot.MeetingUrl.MeetingID, bot.MeetingUrl.Platform)

			database.CacheInstance.Set(context.Background(), "meeting_id_bind-"+meeting.Id, link, time.Hour*24*45)

		}
	}
}

func getMeetingLink(id, platform string) string {
	switch platform {
	case "microsoft_teams":
		return "https://teams.live.com/meet/" + id

	case "google_meet":
		return "https://meet.google.com/" + id
	default:
		return ""
	}
}
